equations{ 'csp.gss.pwrb.temp_par_design' } = define() {
	return ${csp.gss.pwrb.temp_par_f0}
+${csp.gss.pwrb.temp_par_f1}
+${csp.gss.pwrb.temp_par_f2}
+${csp.gss.pwrb.temp_par_f3};
};

equations{ 'csp.gss.pwrb.pl_par_design' } = define() {
	return ${csp.gss.pwrb.pl_par_f0}
+${csp.gss.pwrb.pl_par_f1}
+${csp.gss.pwrb.pl_par_f2}
+${csp.gss.pwrb.pl_par_f3};
};

equations{ 'dni_par_design' } = define() {
	return ${dni_par_f0}
+${dni_par_f1}
+${dni_par_f2}
+${dni_par_f3};
};

equations{ 'W_dot_par_fixed' } = define() {
	return ${w_des}*${f_Wpar_fixed};
};

equations{ 'csp.gss.pwrb.design_parasitic_load' } = define() {
	return ${w_des}*${f_Wpar_fixed} + ${w_des}*${f_Wpar_prod}*
			(
				${csp.gss.pwrb.pl_par_design} +
				${csp.gss.pwrb.temp_par_design} + 
				${dni_par_design}
			);
};

equations{ 'W_dot_part_load_des' } = define() {
	return ${w_des}*${f_Wpar_prod}*${f_par_tot_des};
};

equations{ 'f_par_tot_des' } = define()
{
	return ${csp.gss.pwrb.pl_par_design} +
				${csp.gss.pwrb.temp_par_design} + 
				${dni_par_design};
};

equations{ 'csp.gss.pwrb.nameplate' } = define() {
	return ${csp.gss.pwrb.gross_net_conversion_factor}
	*${w_des};
};

equations{ 'PC_T_corr' } = define() {
	return ${csp.gss.pwrb.temp_corr_mode}+1;
};

	arr[4] = ${csp.gss.pwrb.pl_eff_f4};
