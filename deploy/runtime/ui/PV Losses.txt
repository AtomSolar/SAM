function pv_losses_enable_subarray_items( isub, en )
{
	prefix = "subarray" + to_string(isub) + "_";
	vars = [ "dcwiring_loss",
		"diodeconn_loss",
		"mismatch_loss",
		"soiling",
		"nameplate_loss",
		"soiling_annual_average",
		"dcloss",
		"tracking_loss",
		"monthly_soiling" ];	
	for (i=0;i<#vars;i++)
		enable( prefix + vars[i], en );
}
