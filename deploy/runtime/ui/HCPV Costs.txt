equations{ 'hcpv.cost.module.power' } = define() {
	return ${hcpv.module.power}*0.001; };

equations{ 'hcpv.cost.modulearray.area' } = define() {
	return ${array_num_trackers}*${array_modules_per_tracker}*${hcpv.module.area}; };

equations{ 'hcpv.cost.contingency' } = define() {
	return ${hcpv.cost.contingency_percent}/100*(
	 ${hcpv.cost.module.totalcost}
	+${hcpv.cost.inverter.totalcost}
	+${hcpv.cost.tracker.totalcost}
	+${hcpv.cost.bos_equip.totalcost}
	+${hcpv.cost.install_labor.totalcost} 
	+${hcpv.cost.install_margin.totalcost} ); };

equations{ 'hcpv.cost.landprep.total' } = define() {
	return  ${hcpv.cost.landprep.per_acre}*${hcpv.cost.land_area.value} 
+ ${hcpv.cost.landprep.percent}/100*${total_direct_cost} 
+ ${hcpv.cost.modulearray.power}*1000*${hcpv.cost.landprep.per_watt} 
+ ${hcpv.cost.landprep.fixed}; };

equations{ 'hcpv.cost.inverter.power' } = define() {
	return 0.001*${inv_snl_paco}; };

equations{ 'hcpv.cost.modulearray.power' } = define() {
	return ${hcpv.cost.module.power}*${hcpv.cost.module.num_units}; };

equations{ 'hcpv.cost.inverter.totalcost' } = define() {
	return ?${hcpv.cost.inverter.costunits}
[
	(${hcpv.cost.inverter.num_units}*${hcpv.cost.inverter.power})*1000,
	${hcpv.cost.inverter.num_units}
]
*${hcpv.cost.per_inverter}; };

equations{ 'hcpv.cost.inverterarray.power' } = define() {
	return ${hcpv.cost.inverter.power}*${hcpv.cost.inverter.num_units}; };

equations{ 'hcpv.cost.module.totalcost' } = define() {
	return ?${hcpv.cost.module.costunits}
[
	(${hcpv.cost.module.num_units}*${hcpv.cost.module.power})*1000,
	${hcpv.cost.module.num_units}
]
*${hcpv.cost.per_module}; };

equations{ 'hcpv.cost.land.total' } = define() {
	return ${hcpv.cost.land.per_acre}*${hcpv.cost.land_area.value} 
+ ${hcpv.cost.land.percent}/100*${total_direct_cost} 
+ ${hcpv.cost.modulearray.power}*1000*${hcpv.cost.land.per_watt} 
+ ${hcpv.cost.land.fixed}; };

equations{ 'total_indirect_cost' } = define() {
	return ${hcpv.cost.permitting.total}
+${hcpv.cost.engr.total}
+${hcpv.cost.grid.total}
+${hcpv.cost.land.total}
+${hcpv.cost.landprep.total}
+${hcpv.cost.sales_tax.total}; };

equations{ 'hcpv.cost.permitting.total' } = define() {
	return ${hcpv.cost.permitting.percent}/100*${total_direct_cost} 
+ ${hcpv.cost.modulearray.power}*1000*${hcpv.cost.permitting.per_watt} 
+ ${hcpv.cost.permitting.fixed}; };

equations{ 'hcpv.cost.installed_per_capacity' } = define() {
	return 0.001*${total_installed_cost}/${hcpv.cost.modulearray.power}; };

equations{ 'hcpv.cost.install_margin.totalcost' } = define() {
	return ${hcpv.cost.install_margin_fixed}
+ ${hcpv.cost.modulearray.power}*1000*${hcpv.cost.install_margin_perwatt}
+ ${hcpv.cost.modulearray.area}*${hcpv.cost.install_margin_perarea}; };

equations{ 'hcpv.cost.land_area.value' } = define() {
	return  ${hcpv.array.total_land_area}; };

equations{ 'hcpv.cost.install_labor.totalcost' } = define() {
	return ${hcpv.cost.install_labor_fixed}
+ ${hcpv.cost.modulearray.power}*1000*${hcpv.cost.install_labor_perwatt}
+ ${hcpv.cost.modulearray.area}*${hcpv.cost.install_labor_perarea}; };

equations{ 'hcpv.cost.grid.total' } = define() {
	return ${hcpv.cost.grid.percent}/100*${total_direct_cost} 
+ ${hcpv.cost.modulearray.power}*1000*${hcpv.cost.grid.per_watt} 
+ ${hcpv.cost.grid.fixed}; };

equations{ 'hcpv.cost.tracker.totalcost' } = define() {
	return ${hcpv.cost.tracker_fixed}
+ ${hcpv.cost.modulearray.power}*1000*${hcpv.cost.tracker_perwatt}
+ ${hcpv.cost.modulearray.area}*${hcpv.cost.tracker_perarea}; };

equations{ 'hcpv.cost.sales_tax.value' } = define() {
	return ${sales_tax_rate}; };

equations{ 'hcpv.cost.module.num_units' } = define() {
	return ${array_num_trackers}*${array_modules_per_tracker}; };

equations{ 'hcpv.cost.sales_tax.total' } = define() {
	return ${hcpv.cost.sales_tax.value}/100
*${total_direct_cost}
*${hcpv.cost.sales_tax.percent}/100; };

equations{ 'hcpv.cost.inverter.num_units' } = define() {
	return ${array_num_inverters}; };

equations{ 'total_direct_cost' } = define() {
	return ${hcpv.cost.module.totalcost}
+${hcpv.cost.inverter.totalcost}
+${hcpv.cost.tracker.totalcost}
+${hcpv.cost.bos_equip.totalcost}
+${hcpv.cost.install_labor.totalcost}
+${hcpv.cost.install_margin.totalcost}
+${hcpv.cost.contingency}; };

equations{ 'hcpv.cost.bos_equip.totalcost' } = define() {
	return ${hcpv.cost.bos_equip_fixed}
+ ${hcpv.cost.modulearray.power}*1000*${hcpv.cost.bos_equip_perwatt}
+ ${hcpv.cost.modulearray.area}*${hcpv.cost.bos_equip_perarea}; };

equations{ 'total_installed_cost' } = define() {
	return ${total_direct_cost}+${total_indirect_cost}; };

equations{ 'hcpv.cost.engr.total' } = define() {
	return ${hcpv.cost.engr.percent}/100*${total_direct_cost} 
+ ${hcpv.cost.modulearray.power}*1000*${hcpv.cost.engr.per_watt} 
+ ${hcpv.cost.engr.fixed}; };

equations { 'system_use_lifetime_output' } = define () {
	return 0;};
equations { 'system_use_recapitalization' } = define () {
	return 0;};
